```jsx
import { Button, T, Spinner } from '@kaynora/ui'
import { useState } from 'react'

export default function () {
  const [state, setState] = useState(true)

  return (
    <div style={{
      display: 'flex',
      flexFlow: 'row nowrap',
      gap: '16px',
      alignItems: 'center',
    }}>
      <Spinner state={state} />
      <Button onClick={() => setState(!state)}>
        <T>Toggle state</T>
      </Button>
    </div>
  )
}
```
