```jsx
import { Button, T, Visibility } from '@kaynora/ui'
import { useState } from 'react'

export default function () {
  const [state, setState] = useState(false)

  return (
    <div style={{
      display: 'flex',
      flexFlow: 'row nowrap',
      gap: '16px',
      alignItems: 'center',
    }}>
      <Visibility state={state} />
      <Button onClick={() => setState(!state)}>
        <T>Toggle state</T>
      </Button>
    </div>
  )
}
```
